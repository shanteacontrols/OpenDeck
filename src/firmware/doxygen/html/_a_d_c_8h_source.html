60 inline void setUpADC(adcConf configuration)
61 {
62  ADMUX = 0x00;
63  ADCSRA = 0x0;
64 
65  switch(configuration.prescaler)
66  {
67  //setup adc prescaler
68  case 128:
69  ADCSRA |= (1<<ADPS2)|(1<<ADPS1)|(1<<ADPS0);
70  break;
71 
72  case 64:
73  ADCSRA |= (1<<ADPS2)|(1<<ADPS1);
74  break;
75 
76  case 32:
77  ADCSRA |= (1<<ADPS2)|(1<<ADPS0);
78  break;
79 
80  case 16:
81  ADCSRA |= (1<<ADPS2);
82  break;
83 
84  default:
85  //128 as an fallback
86  ADCSRA |= (1<<ADPS2)|(1<<ADPS1)|(1<<ADPS0);
87  break;
88  }
89 
90  switch(configuration.vref)
91  {
92  case ADC_VREF_AREF:
93  //nothing, this is default setting
94  break;
95 
96  case ADC_VREF_AVCC:
97  ADMUX |= (1<<REFS0);
98  break;
99 
100  case ADC_VREF_INTERNAL_2V56:
101  ADMUX |= (1<<REFS0) | (1<<REFS1);
102  break;
103 
104  case ADC_VREF_INTERNAL_1V1:
105  ADMUX |= (1<<REFS1);
106  break;
107 
108  default:
109  //nothing
110  break;
111  }
112 
113  //enable ADC
114  ADCSRA |= (1<<ADEN);
115 }
